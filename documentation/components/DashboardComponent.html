import { Component, OnInit } from '@angular/core';
import { MediaObserver } from '@angular/flex-layout';
import { DashboardService } from '../../services/dashboard.service';
import { FundraiserCardService } from 'src/app/shared/services/fundraiser-card.service';
import { APIService } from 'src/app/global/services/api.service';
import { Router } from '@angular/router';
import { Subscription } from 'rxjs';
import { CustomBrandingService } from '../../../custom-branding/services/custom-branding.service';
import { BankService } from '../../../profile/services/bank.service';
import { ProfileService } from '../../../profile/services/profile.service';
import { BalanceService } from '../../../balance/balance.service';
import { HttpParams } from '@angular/common/http';
import { AccountService } from 'src/app/pages/account/services/account.service';

@Component({
	selector: 'app-dashboard',
	templateUrl: './dashboard.component.html',
})
/** *Dashboard Component */
export class DashboardComponent implements OnInit {
	profileResponse: any;
	balanceResponse: any;
	myFundraisersResponse: any;
	donationGivenResponse: any;
	donationReceivedResponse: any;
	donationRecurringReceivedResponse: any;
	donationRecurringGivenResponse: any;

	stepsToDoObj: any;
	profileCardObj: any;
	myFundraisersObj!: any;
	myFundraisersCount!: number;
	donationReceivedObj!: any;
	donationGivenObj!: any;
	donationRecurringReceivedObj: any;
	donationRecurringGivenObj: any;
	subscription: Subscription | any;
	oldBalanceTotal: number = 0;
	stripeNotificationCheck: boolean = true;
	stripeStatus: any = {};
	parentFundraiser!: number;
	constructor(
		public _dashboardService: DashboardService,
		public _APIService: APIService,
		private _fundraiserCardService: FundraiserCardService,
		public _media: MediaObserver,
		private _router: Router,
		private _customBrandingService: CustomBrandingService,
		public _profileService: ProfileService,
		public _balanceService: BalanceService,
		public _bankService: BankService,
		public _accountService: AccountService
	) {}

	ngOnInit(): void {
		if (this._accountService.checkHeaders()) {
			/** *Get Stripe Verification Check to Show Notification Bar */
			this._bankService.getStripeStatus().subscribe((res: any) => {
				this.stripeStatus = res?.data;
				//Checking that either user is verified with Stripe or not
				if (
					this.stripeStatus?.charges_enabled == true &&
					this.stripeStatus?.payout_enabled == true &&
					this.stripeStatus?.details_submitted == true
				) {
					this.stripeNotificationCheck = true;
				} else {
					this.stripeNotificationCheck = false;
				}
			});
			this.setDashboardData();
			console.log('the profile card', this.profileCardObj);

			this._dashboardService
				.getFundraiserSummary()
				.subscribe((response: any) => {
					this.parentFundraiser = response.data.parent;
				});
		}
	}

	//check if obj or obj.donationReceived is undefined and then use an optional chaining operator when accessing the donationReceived property to avoid the TypeError if it is undefined
	removeUndefined(obj: any) {
		if (!obj || !obj.donationReceived) {
			return obj;
		}
		let x = obj.donationReceived?.filter(function (val: any) {
			return val.status !== 'undefined';
		});
		obj.donationReceived = x;
		return obj;
	}

	/** *Set Dashboard data function, 4 UI component are populated with API responses */
	setDashboardData() {
		if (this._accountService.checkHeaders()) {
			/** *(1) Profile card */
			/** *Get profile for profile card */

			this._dashboardService.getProfile().subscribe(
				(res: any) => {
					/** *Success */
					this.profileResponse = res.data;
					console.log('profile response', this.profileResponse);
					this._customBrandingService.setIsReceived(
						this.profileResponse?.profile?.is_receiver
					);

					this._balanceService.getTotalBalance().subscribe(
						(res: any) => {
							/** *Success */
							/** *filter profile card data from profile and balance response data */
							this.balanceResponse =
								res?.data?.balance?.amount / 100 +
								res?.data?.balance?.amount_reserved / 100;
							console.log('BALANCE RESOPONSE', this.balanceResponse);
							this.profileCardObj =
								this._dashboardService.filterProfileCardData(
									this.profileResponse,
									this.balanceResponse
								);
							this._dashboardService.changeObj(this.profileCardObj);
						},
						(error) => {
							/** *Error */
							this._APIService.handleError(error).subscribe(
								() => {},
								(errorMessage) => {
									console.log('ERROR', errorMessage);
								}
							);
						}
					);

					/** *Get fundraisers */
					this._dashboardService
						.getMyfundraisers(
							new HttpParams({
								fromObject: {
									['status']: 'latest',
									['filter']: '',
									['page']: 0,
									['page_size']: 2,
								},
							})
						)
						.subscribe(
							(res: any) => {
								/** *Success */
								if (res.errors.code == '4001') {
									this.myFundraisersObj = {};
								} else {
									this.myFundraisersResponse = res?.data?.results;
									/** *(2) My Fundraisers */
									this.myFundraisersCount = res?.data?.count;
									if (this.myFundraisersCount == 0) {
										this._dashboardService
											.getMyfundraisers(
												new HttpParams({
													fromObject: {
														['status']: 'draft',
														['filter']: '',
														['page']: 0,
														['page_size']: 2,
													},
												})
											)
											.subscribe(
												(res: any) => {
													/** *Success */
													this.myFundraisersResponse = res?.data?.results;
													/** *(2) My Fundraisers */
													this.myFundraisersCount = res?.data?.count;
													this.myFundraisersObj =
														this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
															this.myFundraisersResponse
														);
													console.log('MY FUNDRIA', this.myFundraisersObj);
												},
												(error) => {
													/** *Error */
													this._APIService.handleError(error).subscribe(
														() => {},
														(errorMessage) => {
															console.log('ERROR', errorMessage);
														}
													);
												}
											);
									}
									this.myFundraisersObj =
										this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
											this.myFundraisersResponse
										);

									/** *(3.1) Donation Summary - Received */
									this._dashboardService.getDonationReceived().subscribe(
										(res) => {
											/** *Success */
											this.donationReceivedResponse = res;
											this.donationReceivedObj =
												this._dashboardService.getDonationReceivedObj(
													this.donationReceivedResponse
												);
											this.donationReceivedObj = this.removeUndefined(
												this.donationReceivedObj
											);
											console.log('DONATION', this.donationReceivedObj);
										},
										(error) => {
											/** *Error */
											this._APIService.handleError(error).subscribe(
												() => {},
												(errorMessage) => {
													console.log('ERROR', errorMessage);
												}
											);
										}
									);
									/** *(3.2) Donation Summary - Given */
									this._dashboardService.getDonationGiven().subscribe(
										(res) => {
											/** *Success */
											this.donationGivenResponse = res;
											this.donationGivenObj =
												this._dashboardService.getDonationGivenObj(
													this.donationGivenResponse
												);

											this.stepsToDoObj =
												this._dashboardService.filterStepsToDoData(
													this.profileResponse,
													this.myFundraisersResponse,
													this.donationGivenResponse
												);
										},
										(error) => {
											/** *Error */
											this._APIService.handleError(error).subscribe(
												() => {},
												(errorMessage) => {
													console.log('ERROR', errorMessage);
												}
											);
										}
									);

									/** *(3.3) Donation Summary - Recurring Received */
									this._dashboardService
										.getRecurringReceivedDonation()
										.subscribe(
											(res) => {
												/** *Success */
												this.donationRecurringReceivedResponse = res;
												console.log(
													'DOOOOOO1',
													this.donationRecurringReceivedResponse
												);

												this.donationRecurringReceivedObj =
													this._dashboardService.getRecurringReceivedDonationObj(
														this.donationRecurringReceivedResponse
													);
											},
											(error) => {
												console.log('ERROR', error);
												/** *Error */
												this._APIService.handleError(error).subscribe(
													() => {},
													(errorMessage) => {
														console.log('ERROR', errorMessage);
													}
												);
											}
										);

									/** *(3.4) Donation Summary - Recurring Given */
									this._dashboardService.getRecurringGivenDonation().subscribe(
										(res) => {
											/** *Success */
											this.donationRecurringGivenResponse = res;
											console.log(
												'DOOOOOO',
												this.donationRecurringGivenResponse
											);

											this.donationRecurringGivenObj =
												this._dashboardService.getRecurringGivenDonationObj(
													this.donationRecurringGivenResponse
												);
										},
										(error) => {
											console.log('ERROR', error);
											/** *Error */
											this._APIService.handleError(error).subscribe(
												() => {},
												(errorMessage) => {
													console.log('ERROR', errorMessage);
												}
											);
										}
									);
								}
							},
							(error) => {
								console.log('ERROR', error);
								/** *Error */
								this._APIService.handleError(error).subscribe(
									() => {},
									(errorMessage) => {
										console.log('ERROR', errorMessage);
									}
								);
							}
						);
				},
				(error) => {
					this._APIService.handleError(error).subscribe(
						() => {},
						(errorMessage) => {
							console.log('ERROR', errorMessage);
						}
					);
				}
			);
		}
	}
}
