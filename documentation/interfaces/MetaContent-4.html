import { Inject, Injectable, LOCALE_ID } from '@angular/core';
import { MetaTagService } from 'src/app/global/services/meta-tag.service';
import { environment } from 'src/environments/environment';
import { AccountService } from 'src/app/pages/account/services/account.service';
import { LinkService } from 'src/app/global/services/link.service';
interface MetaContent {
	title: string;
	description: string;
	url: string;
}

@Injectable({
	providedIn: 'root',
})
/** *Search Meta Data Service */
export class SearchMetaDataService {
	HOME_URL: string = environment.homeUrl;
	meta: MetaContent = {
		title: $localize`:@@search_meta_data_searchFundraiser_title: Search Fundraisers | Whydonate`,
		description: $localize`:@@search_meta_data_feelingGenerous_description: Feeling generous? Discover your favourite fundraiser on Whydonate, the best crowdfunding platform for individuals and NGOs.`,
		url: `search`,
	};
	constructor(
		private _accountService: AccountService,
		@Inject(LOCALE_ID) public locale: string,
		private _metatagService: MetaTagService,
		private _linkService: LinkService
	) {}
	updateSearchMetaTag(): void {
		this._metatagService.setTitle(this.meta.title);
		this._metatagService.updateDescription(this.meta.description);
		/** *og meta tag */
		this._metatagService.updateOgMeta({
			'og:title': this.meta.title,
			'og:description': this.meta.description,
			'og:url': this.getMetaUrl(this.meta.url),
			'og:type': 'website',
			'og:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1.jpg',
			'og:image:width': '1200',
			'og:image:height': '627',
		});
		/** *twitter meta tag */
		this._metatagService.updateTwitterMeta({
			'twitter:title': this.meta.title,
			'twitter:description': this.meta.description,
			'twitter:url': this.getMetaUrl(this.meta.url),
			'twitter:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1.jpg',
			'twitter:card': 'summary_large_image',
		});

		// Links
		this._linkService.removeAlternateLinkTags();
		this._linkService.getAltLinks(this.meta.url).forEach((element) => {
			this._linkService.addAlternateLinkTags(element);
		});
		this._linkService.addAlternateLinkTags(
			this._linkService.getDefaultLink(this.meta.url)
		);
	}
	/** page Url */
	getMetaUrl(url: string) {
		return `${this.HOME_URL}/${this._accountService.getLocaleId()}/${url}`;
	}
}
