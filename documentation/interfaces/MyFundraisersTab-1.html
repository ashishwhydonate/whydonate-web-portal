	setStepsToDo() {
		/** *(1) Profile card */
		this._dashboardService.getProfile().subscribe(
			(res: any) => {
				/** *Success */
				this.profileResponse = res.data;

				/** *(2) Get fundraisers */
				this._dashboardService
					.getMyfundraisers(
						new HttpParams({
							fromObject: {
								['status']: 'published',
								['filter']: '',
								['page']: 0,
								['page_size']: 2,
							},
						})
					)
					.subscribe(
						(res: any) => {
							/** *Success */
							this.myFundraisersResponse = res?.data?.results;
							/** *(3) Donation Summary - Given */
							this._dashboardService.getDonationGiven().subscribe(
								(res) => {
									/** *Success */
									this.donationGivenResponse = res;

									/** *(4) Steps To Do */

									this.stepsToDoObj =
										this._dashboardService.filterStepsToDoData(
											this.profileResponse,
											this.myFundraisersResponse,
											this.donationGivenResponse
										);
								},
								(error) => {
									this._APIService.handleError(error).subscribe(
										() => {},
										(errorMessage) => {
											this._router.navigate(['account']);
										}
									);
								}
							);
						},
						(error) => {
							this._APIService.handleError(error).subscribe(
								() => {},
								(errorMessage) => {
									this._router.navigate(['account']);
								}
							);

					() => {},
					(errorMessage) => {
						this._router.navigate(['account']);
					}

	ngAfterViewInit(): void {
		this.filterOwnershipForm.valueChanges.subscribe((data: any) => {
			if (
				(data.not_connected == true && data.connected == true) ||
				(data.connected == false && data.not_connected == false)
			) {
				this.both();
			}
			if (data.connected == true && data.not_connected == false) {
				this.connectedType();
			}
			if (data.connected == false && data.not_connected == true) {
				this.mainType();
			}

			this.setFundraiserTypePayload(data);
		});
		/** *Subscribed to search input on valuechange event. Added pipe to optimsed the API call */
		this.searchInputForm?.controls.searchCtrl.valueChanges
			.pipe(
				debounce(() => {
					return interval(this.debounceDelay);
				}),
				distinctUntilChanged((x, y) => {
					/** *INFO: If no actual change then cancel the loading and Search */
					if (x === y) {
						// console.log(`query x: ${x}, y: ${y} are identical, api called interrupted`);
						this.cancelLoadingAndProgress();
						return true;
					}
					// if (this.isLoading && this.progressBarMode === 'query') {
					// 	//TODO: If true then, new API request made while one is already running. (user switchMap from rxjs to resolve this scenario: https://www.learnrxjs.io/learn-rxjs/operators/transformation/switchmap)
					// }
					return false;
				})
			)
			.subscribe((data) => {
				/** *loading flag and progress bar setting */
				this.startTabsLoading();
				this.startCountLoading();
				this.startQueryProgress();

				// console.log('native fromControl value changes with debounce: ', data);
				/** *INFO: If data is null or undefined then assign emtry param */
				this._myFundraisersService.setFilterString = data || '';
				this.filter = data;
				// console.log(this._myFundraisersService.getCountParams);
				let filteredCountParams = this._myFundraisersService.getCountParams;
				// Count
				this._myFundraisersService
					.getCount(filteredCountParams)
					.subscribe((response: any) => {
						this.fundraisersTotalCount = response.data.count;
						this.isCountLoading = false;
						this.finishCountLoading();
					});

				// Tab data
				// TODO: When user search, There can be scenarios when one request already running and another request is made, experience can be buggy. possible solution: load all tabs at once for the request coming from search.
				//! Below code shows empty tabs so when requests race with each other, the order of tab is not changed and no duplicate tabs. can be fixed by approach mentioned in above comment.
				this.fundraiserDataList = [
					{
						label: $localize`:@@myFundraisers_drafts_mat_tab:Drafts`,
						count: 0,
						content: [],
					},
					{
						label: $localize`:@@myFundraisers_published_mat_tab:Published`,
						count: 0,
						content: [],
					},
					{
						label: $localize`:@@myFundraisers_closed_mat_tab:Closed`,
						count: 0,
						content: [],
					},
				];

				this._myFundraisersService.getAll().subscribe(
					(response: any) => {
						this.setProgressbar(response.label);

						if (response && response?.label) {
							const existingTab = this.fundraiserDataList?.find((tab) => {
								if (tab !== undefined) {
									return tab.label === response?.label;
								}
							});
							if (existingTab) {
								existingTab.content = response?.content;
								existingTab.count = response.content.length;
								if (this.filter !== '') {
									if (existingTab.label == 'Drafts')
										this.draftCount = response.content.length;
									if (existingTab.label == 'Published')
										this.publishedCount = response.content.length;
									if (existingTab.label == 'Closed')
										this.closedCount = response.content.length;
								} else {
									if (existingTab.label == 'Drafts')
										this.draftCount = response.count;
									if (existingTab.label == 'Published')
										this.publishedCount = response.count;
									if (existingTab.label == 'Closed')
										this.closedCount = response.count;
								}
							} else {
								this.fundraiserDataList?.push(response?.results);
							}
						}
					},
					(error) => {
						this._APIService.handleError(error).subscribe(
							() => {},
							(errorMessage) => {
								this._router.navigate(['account']);
							}
						);
					}
				);
			});
	}

	viewMore() {
		this.temp_not_connected =
			this.filterOwnershipForm.getRawValue().not_connected;
		this.temp_connected = this.filterOwnershipForm.getRawValue().connected;
		console.log('TEMNOT', this.temp_not_connected);
		console.log('TEM', this.temp_connected);

		if (
			(this.temp_connected === true && this.temp_not_connected === true) ||
			(this.temp_connected === false && this.temp_not_connected === false)
		) {
			this.type = 'both';
			console.log('IF');
		} else if (
			this.temp_not_connected === true &&
			this.temp_connected === false
		) {
			this.type = 'not_connected';
			console.log('ELSEIF');
		} else if (
			this.temp_not_connected === false &&
			this.temp_connected === true
		) {
			this.type = 'connected';
			console.log('ELSE');
		}

		let label: string = '';

		switch (this.selectedTab) {
			case 0:
				label = 'Drafts';
				break;
			case 1:
				label = 'Published';
				break;
			case 2:
				label = 'Closed';
				break;
			default:
				break;
		}
		let limit = 0;
		if (this.selectedTab == 0) {
			this.pageLimit.drafts = this.pageLimit.drafts + 5;
			limit = this.pageLimit.drafts;
		} else if (this.selectedTab == 1) {
			this.pageLimit.published = this.pageLimit.published + 5;
			limit = this.pageLimit.published;
		} else {
			this.pageLimit.closed = this.pageLimit.closed + 5;
			limit = this.pageLimit.closed;
		}
		this.startTabsLoading();
		this.startQueryProgress();
		this._dashboardService
			.getMyfundraisers(
				new HttpParams({
					fromObject: {
						['page']: limit,
						['page_size']: 5,
						['filter']: this.filter,
						['status']:
							(this.selectedTab == 0 && 'draft') ||
							(this.selectedTab == 1 && 'published') ||
							(this.selectedTab == 2 && 'closed'),
						['type']: this.type,
					},
				})
			)
			.subscribe(
				(res: any) => {
					this.myFundraisersResponse = res;
					if (
						res?.data &&
						res?.data?.results &&
						res?.data?.results?.length == 0
					) {
						if (this.selectedTab == 0) {
							this.disableViewMore.drafts = true;
						} else if (this.selectedTab == 1) {
							this.disableViewMore.published = true;
						} else {
							this.disableViewMore.closed = true;
						}
					}
					if (
						this.fundraiserDataList &&
						this.fundraiserDataList.filter((i) => i.label == label)?.length >
							0 &&
						this.fundraiserDataList.filter((i) => i.label == label)?.[0]?.content
					) {
						this.fundraiserDataList
							.filter((i) => i.label == label)[0]
							?.content?.push(
								...this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
									res?.data?.results
								)
							);
					}

					limit = 0;
					this.setProgressbar(res?.label);
					this.finishCountLoading();
					this.cancelLoadingAndProgress();
				},
				(error) => {
					this._APIService.handleError(error).subscribe(
						() => {},
						(errorMessage) => {
							this._router.navigate(['account']);
						}
					);
				}
			);
	}

	setFundraiserTypePayload(formValue: any) {
		if (
			formValue[TYPE_NOT_CONNECTED] === true &&
			formValue[TYPE_CONNECTED] === false
		) {
			this._myFundraisersService.setTypeMain();
			return;
		}
		if (
			formValue[TYPE_NOT_CONNECTED] === true &&
			formValue[TYPE_CONNECTED] === false
		) {
			this._myFundraisersService.setTypeConnected();
			return;
		} else this._myFundraisersService.setTypeBoth();
	}

	// Increase progress bar on getting each tab's data
	setProgressbar(tab: string) {
		if (tab === 'Drafts') {
			this.startDeterminateProgress();
			this.progressbarValue = this.progressbarValue + 30;
			return;
		}
		if (tab === 'Closed') {
			this.finishDeterminateProgress();
			this.isLoading = false;
			return;
		}
		this.progressbarValue = this.progressbarValue + 40;
	}
