import { Injectable } from '@angular/core';
import { Meta, Title } from '@angular/platform-browser';

/**
 * This class takes care of Meta Tags in application pages
 */

@Injectable({
	providedIn: 'root',
})
export class MetaTagService {
	/**
	 * Constructor Depends upon Meta and Title from platform browser
	 * @param _meta
	 * @param _title
	 */
	constructor(private _meta: Meta, private _title: Title) {}

	/**
	 * This function sets page Title
	 * @param page_title
	 */
	setTitle(page_title: string): void {
		try {
			this._title.setTitle(page_title);
		} catch (error) {
			console.error('Error in setTitle function in meta-tag service : ', error);
		}
	}

	/**
	 * This function updates page description
	 * @param meta_description
	 */
	updateDescription(meta_description: string): void {
		try {
			this._meta.updateTag(
				{
					name: 'description',
					content: meta_description,
				},
				"name='description'"
			);
		} catch (error) {
			console.error(
				'Error in updateDescription function in meta-tag service : ',
				error
			);
		}
	}

	/**
	 * This function updates OG Meta
	 * @param og_meta_arr
	 */
	updateOgMeta(og_meta_arr: any): void {
		try {
			for (const key in og_meta_arr) {
				if (Object.prototype.hasOwnProperty.call(og_meta_arr, key)) {
					this._meta.updateTag(
						{
							property: key,
							content: og_meta_arr[key],
						},
						"property='" + key + "'"
					);
				}
			}
		} catch (error) {
			console.error(
				'Error in updateOgMeta function in meta-tag service : ',
				error
			);
		}
	}

	/**
	 * This function updates twitter meta
	 * @param twitter_meta_arr
	 */
	updateTwitterMeta(twitter_meta_arr: any): void {
		try {
			for (const key in twitter_meta_arr) {
				if (Object.prototype.hasOwnProperty.call(twitter_meta_arr, key)) {
					this._meta.updateTag(
						{
							property: key,
							content: twitter_meta_arr[key],
						},
						"property='" + key + "'"
					);
				}
			}
		} catch (error) {
		
		}
	}
}
