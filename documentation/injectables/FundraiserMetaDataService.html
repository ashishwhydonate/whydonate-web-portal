import { Inject, Injectable, LOCALE_ID, SecurityContext } from '@angular/core';
import { DomSanitizer } from '@angular/platform-browser';
import { MetaTagService } from 'src/app/global/services/meta-tag.service';
import { environment } from 'src/environments/environment';
import { LinkService } from 'src/app/global/services/link.service';

interface MetaContent {
	title: string;
	description: string;
	url: string;
	image: string;
}

@Injectable({
	providedIn: 'root',
})
/** *Fundraiser Meta Data Service */
export class FundraiserMetaDataService {
	HOME_URL: string = environment.homeUrl;

	// Keyword for SEO
	metaKeyword = $localize`:@@fundraiser_meta_data_fundraising_keyword: Fundraising`;

	constructor(
		private _metatagService: MetaTagService,
		private domSanitizer: DomSanitizer,
		@Inject(LOCALE_ID) public locale: string,
		private _linkService: LinkService
	) {}
	/** *Function to update meta tags from currentFundraiser */
	updateFundraiserMetaTag(currentFundraiser: any): void {
		console.log(currentFundraiser);
		let metaDataObj = this.filterFundraiserMetaData(currentFundraiser);

		/** *page title */
		this._metatagService.setTitle(metaDataObj.title);
		/** *description meta tag */
		this._metatagService.updateDescription(metaDataObj.description);
		/** *og meta tag */
		this._metatagService.updateOgMeta({
			'og:title': metaDataObj.title,
			'og:description': metaDataObj.description,
			'og:url': metaDataObj.url,
			'og:type': 'website',
			'og:image': metaDataObj.image,
			'og:image:width': '1200',
			'og:image:height': '627',
		});
		/** *twitter meta tag */
		this._metatagService.updateTwitterMeta({
			'twitter:title': metaDataObj.title,
			'twitter:description': metaDataObj.description,
			'twitter:url': metaDataObj.url,
			'twitter:image': metaDataObj.image,
			'twitter:card': 'summary_large_image',
		});

		// Links
		let url = `fundraising/${currentFundraiser?.slug}`;
		this._linkService.removeAlternateLinkTags();
		this._linkService.getAltLinks(url).forEach((element) => {
			this._linkService.addAlternateLinkTags(element);
		});
		this._linkService.addAlternateLinkTags(
			this._linkService.getDefaultLink(url)
		);
	}
	/** *Calls other filter function to get data for MetaContent */
	filterFundraiserMetaData(currentFundraiser: any): MetaContent {
		return {
			title: this.filterMetaTitle(currentFundraiser),
			description: this.filterMetaDescription(currentFundraiser),
			url: this.filterMetaUrl(currentFundraiser),
			image: this.filterMetaImage(currentFundraiser),
		};
	}
	/** *return title like this=> keyword - profile name | title */
	filterMetaTitle(currentFundraiser: any) {
		let fundraising_title = currentFundraiser?.title || '';
		let fundraising_profile_name = currentFundraiser?.profile?.name || '';
		return (
			`${this.metaKeyword} ${fundraising_profile_name}` +
			` | ` +
			`${fundraising_title}`
		).substring(0, 70);
	}
	/** *return description like this=> description | metaKeyword */
	/** *meta description filtered with condition like this=> if "appeal" true, then return fundraising.appeal + metaKeyword, if "content" true, then return fundraising.content + metaDescription, else return metaDescription */
	filterMetaDescription(currentFundraiser: any) {
		let meta_description: string;
		if (currentFundraiser?.appeal) {
			/** *Sanitize and escape html for appeal_string */
			let appeal_string =
				this.domSanitizer.sanitize(
					SecurityContext.HTML,
					currentFundraiser?.appeal
				) || '';
			appeal_string = this.escapeHtml(appeal_string).substring(0, 140).trim();
			meta_description = `${appeal_string} | ${this.metaKeyword}`;
			return meta_description;
		}
		if (currentFundraiser?.content) {
			let content_string =
				this.domSanitizer.sanitize(
					SecurityContext.HTML,
					currentFundraiser?.content
				) || '';
			content_string = this.escapeHtml(content_string).substring(0, 140).trim();
			meta_description = `${content_string} | ${this.metaKeyword}`;
			return meta_description;
		}
		return this.metaKeyword;
	}
	/** *Url to fundraiser page */
	filterMetaUrl(currentFundraiser: any) {
		return `${this.HOME_URL}/${this.getLocale(this.locale)}/fundraising/${
			currentFundraiser?.slug
		}`;
	}
	/** *return background image url from currentFundraiser or return default background */
	filterMetaImage(currentFundraiser: any) {
		let staticMetaImage =
			'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1.jpg';
		return currentFundraiser?.background?.image || staticMetaImage;
	}

	/** *function to escape or replace html tag and double space */
	escapeHtml(string_with_html: any) {
		/** *.(remove any tags).(remove double space) */
		return string_with_html.replace(/<[^>]*>/g, ' ').replace(/\s\s+/g, ' ');
	}

	/** *Get locale for creating route */
	getLocale(locale: string) {
		if (locale.includes('en')) return 'en';
		if (locale.includes('nl')) return 'nl';
		if (locale.includes('de')) return 'de';
		if (locale.includes('es')) return 'es';
		if (locale.includes('fr')) return 'fr';

		/** *default, if notthing found */
		return 'nl';
	}
}
