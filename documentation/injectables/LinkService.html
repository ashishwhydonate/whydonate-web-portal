import { DOCUMENT } from '@angular/common';
import { Inject, Injectable, Renderer2, RendererFactory2 } from '@angular/core';
import { environment } from 'src/environments/environment';

@Injectable({
	providedIn: 'root',
})
export class LinkService {
	allLocales = ['en', 'nl', 'es', 'de', 'fr'];
	HOME_URL: string = environment.homeUrl;
	private renderer: Renderer2;
	constructor(
		@Inject(DOCUMENT) private _document: Document,
		rendererFactory: RendererFactory2
	) {
		this.renderer = rendererFactory.createRenderer(null, null);
	}

	removeAlternateLinkTags() {
		try {
			const head = this._document.head;
			if (head === null) {
				throw new Error('<head> not found within DOCUMENT.');
			}
			// get <link rel='alternate' href='' hreflang=''> based on hreflang
			const eleHref = this._document.querySelectorAll('link[rel="alternate"]');
			eleHref.forEach(function (elem) {
				elem.remove();
			});
		} catch (e) {
			console.error('Error within removeAlternateLinkTags : ', e);
		}
	}

	addAlternateLinkTags(tag: LinkDefinition) {
		try {
			const link = this.renderer.createElement('link');
			const head = this._document.head;

			if (head === null) {
				throw new Error('<head> not found within DOCUMENT.');
			}

			Object.keys(tag).forEach((prop: string) => {
				return this.renderer.setAttribute(link, prop, tag[prop]);
			});

			// [TODO]: get them to update the existing one (if it exists) ?
			this.renderer.appendChild(head, link);
		} catch (e) {
			console.error('Error within linkService : ', e);
		}
	}

	getAltLinks(url: string): LinkDefinition[] {
		return this.allLocales.map((x) => {
			return {
				rel: 'alternate',
				hreflang: x,
				href: `${this.HOME_URL}/${x}/${url}`,
			};
		});
	}
	getDefaultLink(url: string): LinkDefinition {
		return {
			rel: 'alternate',
			hreflang: 'x-default',
			href: `${this.HOME_URL}/${url}`,
		};
	}
}

export declare type LinkDefinition = {
	charset?: string;
	crossorigin?: string;
	href?: string;
	hreflang?: string;
	media?: string;
	rel?: string;
	rev?: string;
	sizes?: string;
	target?: string;
	type?: string;
} & {
	[prop: string]: string;
};
