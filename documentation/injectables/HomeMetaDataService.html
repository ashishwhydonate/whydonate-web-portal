import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { MetaTagService } from '../../../global/services/meta-tag.service';
import { AccountService } from 'src/app/pages/account/services/account.service';
import { LinkService } from 'src/app/global/services/link.service';

// type for key for meta
type HomeMeta = 'personal' | 'organisation';
// type for value for meta
interface MetaContent {
	title: string;
	description: string;
	url: string;
}

@Injectable({
	providedIn: 'root',
})
/** *Home Meta Data Service*/
export class HomeMetaDataService {
	HOME_URL: string = environment.homeUrl;
	meta: Record<HomeMeta, MetaContent> = {
		personal: {
			title: $localize`:@@home_meta_data_personalCrowdFunding_title: WhyDonate: Personal Crowdfunding and Charity Fundraising`,
			description: $localize`:@@home_meta_data_whyDonateIsBestCrowdFunding_description: WhyDonate is the Best Crowdfunding Platform and Fundraising Site for Charity and Personal use. Raise Money for Charity and Individuals for Free!`,
			url: ``,
		},
		organisation: {
			title: $localize`:@@home_meta_data_charityFundraising_title: Charity Fundraising & Best Crowdfunding site for Charity`,
			description: $localize`:@@home_meta_data_organisationCrowdFunding_description: Whydonate is the Best Fundraising and Crowdfunding site for Charity. Raise money for charity on Whydonate for free!`,
			url: `organisation`,
		},
	};

	constructor(
		private _accountService: AccountService,
		private _metatagService: MetaTagService,
		private _linkService: LinkService
	) {}

	updatePersonalMetaTag(): void {
		/** *set page title and update decription meta tag */
		this._metatagService.setTitle(this.meta.personal.title);

		this._metatagService.updateDescription(this.meta.personal.description);
		/** *og meta tag */
		this._metatagService.updateOgMeta({
			'og:title': this.meta.personal.title,
			'og:description': this.meta.personal.description,
			'og:url': this.getMetaUrl(this.meta.personal.url),
			'og:type': 'website',
			'og:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1',
			'og:image:width': '1200',
			'og:image:height': '627',
		});
		/** *twitter meta tag */
		this._metatagService.updateTwitterMeta({
			'twitter:title': this.meta.personal.title,
			'twitter:description': this.meta.personal.description,
			'twitter:url': this.getMetaUrl(this.meta.personal.url),
			'twitter:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1',
			'twitter:card': 'summary_large_image',
		});

		// Links
		this._linkService.removeAlternateLinkTags();
		this._linkService.getAltLinks(this.meta.personal.url).forEach((element) => {
			this._linkService.addAlternateLinkTags(element);
		});
		this._linkService.addAlternateLinkTags(
			this._linkService.getDefaultLink(this.meta.personal.url)
		);
	}

	updateOrganisationMetaTag(): void {
		/** *set page title and update decription meta tag */
		this._metatagService.setTitle(this.meta.organisation.title);
		this._metatagService.updateDescription(this.meta.organisation.description);
		/** *og meta tag */
		this._metatagService.updateOgMeta({
			'og:title': this.meta.organisation.title,
			'og:description': this.meta.organisation.description,
			'og:url': this.getMetaUrl(this.meta.organisation.url),
			'og:type': 'website',
			'og:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1',
			'og:image:width': '1200',
			'og:image:height': '627',
		});
		/** *twitter meta tag */
		this._metatagService.updateTwitterMeta({
			'twitter:title': this.meta.organisation.title,
			'twitter:description': this.meta.organisation.description,
			'twitter:url': this.getMetaUrl(this.meta.organisation.url),
			'twitter:image':
				'https://res.cloudinary.com/whydonate/image/upload/dpr_auto,f_auto,q_auto/facebook_1',
			'twitter:card': 'summary_large_image',
		});

		// Links
		this._linkService.removeAlternateLinkTags();
		this._linkService
			.getAltLinks(this.meta.organisation.url)
			.forEach((element) => {
				this._linkService.addAlternateLinkTags(element);
			});
		this._linkService.addAlternateLinkTags(
			this._linkService.getDefaultLink(this.meta.organisation.url)
		);
	}

	/** page Url */
	getMetaUrl(url: string) {
		return `${this.HOME_URL}/${this._accountService.getLocaleId()}/${url}`;
	}
}
