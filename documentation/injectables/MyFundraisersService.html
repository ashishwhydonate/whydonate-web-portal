import { HttpClient, HttpHeaders, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { APIService } from 'src/app/global/services/api.service';
import { AccountService } from 'src/app/pages/account/services/account.service';
import { environment } from 'src/environments/environment';
import { Observable, Observer } from 'rxjs';
import { publishReplay, refCount, tap } from 'rxjs/operators';
import { FundraiserCardService } from 'src/app/shared/services/fundraiser-card.service';
export interface MyFundraisersTab {
	label: string;
	count: number;
	content: any[];
}
/** *Constant string lateral */
export const TYPE_NOT_CONNECTED = 'not_connected';
export const TYPE_CONNECTED = 'connected';
export const TYPE_BOTH = 'connected';

@Injectable({
	providedIn: 'root',
})
/** *My Fundraisers Service */
export class MyFundraisersService {
	FUNDRAISER_URL: string = environment.fundraiser_url;
	private readonly userFundraiserAPI =
		'fundraiser/get_user_fundraisers_status/';
	private readonly userFundraiserLocalAPI = 'project/fundraising/local/';
	private readonly userFundraiserOwnerAPI =
		'fundraiser/get_user_fundraisers_status/';
	private readonly defaultParams = new HttpParams({
		fromObject: {
			['status']: 'published',
			['filter']: '',
			['page']: 0,
			['page_size']: 5,
		},
	});
	private readonly tempParams = new HttpParams({
		fromObject: {
			['status']: 'draft',
			['filter']: '',
			['page']: 0,
			['page_size']: 5,
			['type']: 'connected',
		},
	});
	private params: HttpParams;
	private header!: HttpHeaders;

	private count: number = 0;
	private draftFundraisers: any;
	private publishedFundraisers: any;
	private closedFundraisers: any;
	private myFundraisers!: MyFundraisersTab[];

	/** *TODO: remove when using api.service */
	API_URL: string = environment.apiUrl;
	ACCOUNT_API_V2: string = environment.ACCOUNT_API_V2;

	/** *allFundraisersObservable: Observable<MyFundraisersTab>; */
	draftParams: any;
	publishedParams: any;
	closedParams: any;
	constructor(
		private _apiService: APIService,
		public _accountService: AccountService,
		private _fundraiserCardService: FundraiserCardService,
		private http: HttpClient
	) {
		/** *INFO: setting params based on default params */
		this.params = this.defaultParams;
	}

	public get getDefaultParams(): HttpParams {
		return this.defaultParams;
	}
	public get getParams(): HttpParams {
		return this.params;
	}
	public get getDraftParams(): HttpParams {
		return this.params.set('status', 'draft');
	}
	public get getPublishedParams(): HttpParams {
		return this.params.set('status', 'published');
	}
	public get getCompletedParams(): HttpParams {
		return this.params.set('status', 'closed');
	}
	public set setFilterString(filter_string: string) {
		this.params = this.newParams.set('filter', filter_string);
	}
	/** *INFO: Set new time and return the prams object */
	public get newParams() {
		return this.params.set('filter', '');
		// return this.params;
	}

	getDraft(params: HttpParams = this.getDraftParams) {
		this.header = this._accountService.getHeaders() as HttpHeaders;
		return this.http.get(this.FUNDRAISER_URL + this.userFundraiserAPI, {
			headers: this.header,
			params,
		});
		// return this._apiService.get(this.userFundraiserAPI,draftParams,this.header);
	}
	getPublished(params: HttpParams = this.getPublishedParams) {
		this.header = this._accountService.getHeaders() as HttpHeaders;
		return this.http.get(this.FUNDRAISER_URL + this.userFundraiserAPI, {
			headers: this.header,
			params,
		});
		// return this._apiService.get(this.userFundraiserAPI,publishedParams,this.header);
	}
	geClosed(params: HttpParams = this.getCompletedParams) {
		this.header = this._accountService.getHeaders() as HttpHeaders;
		return this.http.get(this.FUNDRAISER_URL + this.userFundraiserAPI, {
			headers: this.header,
			params,
		});
		// return this._apiService.get(this.userFundraiserAPI,ClosedParams,this.header);
	}
	// getOwner(params: HttpParams = this.tempParams) {
	// 	let API_URL = environment.fundraiser_url;
	// 	this.header = this._accountService.getHeaders() as HttpHeaders;
	// 	return this.http.get(API_URL + this.userFundraiserOwnerAPI, {
	// 		headers: this.header,
	// 		params,
	// 	});
	// }
	get getCountParams() {
		return this.getParams.set('page_size', 5);
	}
	// TODO: optimise api from backend
	getCount(params: HttpParams = this.getPublishedParams) {
		this.header = this._accountService.getHeaders() as HttpHeaders;
		return this.http
			.get(this.FUNDRAISER_URL + this.userFundraiserAPI, {
				headers: this.header,
				params,
			})
			.pipe(
				tap((res: any) =>
					console.log('Shared service observable emitted', res)
				),
				publishReplay(),
				refCount()
			);
		// return this._apiService.get(this.userFundraiserAPI,ClosedParams,this.header);
	}
	getAllFundraisersAtOnce(): any {
		// count -> draft -> published -> closed/completed
		this.getCount().subscribe((countResponse: any) => {
			this.count = countResponse.data.count;
			// console.log(this.count);

			this.getDraft().subscribe((response: any) => {
				this.draftFundraisers = response.data;
				// console.log(this.draftFundraisers);

				this.getPublished().subscribe((response: any) => {
					this.publishedFundraisers = response.data;
					// console.log(this.publishedFundraisers);

					this.geClosed().subscribe((response: any) => {
						this.closedFundraisers = response.data;
						// console.log(this.closedFundraisers);

						return new Observable((observer: Observer<MyFundraisersTab[]>) => {
							observer.next([
								{
									label: 'Drafts',
									count: this.draftFundraisers.count,
									content: this.draftFundraisers.results,
								},
								{
									label: 'Published',
									count: this.publishedFundraisers.count,
									content: this.publishedFundraisers.results,
								},
								{
									label: 'Closed',
									count: this.closedFundraisers.count,
									content: this.closedFundraisers.results,
								},
							]);
						});
					});
				});
			});
		});
	}

	/**
	 * DESCRIPTION: Observable which gets API data and returns Observer of type MyFundraisersTab
	 */
	public allFundraisersObservable = new Observable(
		(observer: Observer<MyFundraisersTab>) => {
			this.getDraft().subscribe((draftResponse: any) => {
				observer.next({
					label: 'Drafts',
					count: draftResponse?.data?.count,
					content:
						this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
							draftResponse?.data?.results
						),
				});

				this.getPublished().subscribe((publishedRsponse: any) => {
					observer.next({
						label: 'Published',
						count: publishedRsponse?.data?.count,
						content:
							this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
								publishedRsponse?.data?.results
							),
					});

					this.geClosed().subscribe((closedResponse: any) => {
						observer.next({
							label: 'Closed',
							count: closedResponse?.data?.count,
							content:
								this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
									closedResponse?.data?.results
								),
						});
					});
				});
			});
		}
	);

	/**
	 * DESCRIPTION: Observable which gets API type data and returns Observer of type MyFundraisersTab
	 */
	public filterFundraisersObservable = new Observable(
		(observer: Observer<MyFundraisersTab>) => {
			this.getDraft().subscribe((draftResponse: any) => {
				observer.next({
					label: 'Drafts',
					count: draftResponse?.data?.count,
					content:
						this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
							draftResponse?.data?.results
						),
				});

				this.getPublished().subscribe((publishedRsponse: any) => {
					observer.next({
						label: 'Published',
						count: publishedRsponse?.data?.count,
						content:
							this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
								publishedRsponse?.data?.results
							),
					});

					this.geClosed().subscribe((closedResponse: any) => {
						observer.next({
							label: 'Closed',
							count: closedResponse?.data?.count,
							content:
								this._fundraiserCardService.filterFundraiserCardDataListForMyfundraisers(
									closedResponse?.data?.results
								),
						});
					});
				});
			});
		}
	);

	/**
	 * DESCRIPTION: Observable which gets API data and returns Observer of owner in MyFundraisersTab
	 */
	public mainFundraisersObservable = new Observable(
		(observer: Observer<MyFundraisersTab>) => {
			let locale = this._accountService.getLocaleId();
			this.draftParams = new HttpParams({
				fromObject: {
					['status']: 'draft',
					['filter']: '',
					['page']: 0,
					['page_size']: 5,
					['type']: 'not_connected',
					['language_code']: locale
				},
			});
			this.publishedParams = new HttpParams({
				fromObject: {
					['status']: 'published',
					['filter']: '',
					['page']: 0,
					['page_size']: 5,
					['type']: 'not_connected',
					['language_code']: locale
				},
			});
			this.closedParams = new HttpParams({
				fromObject: {
					['status']: 'closed',
					['filter']: '',
					['page']: 0,
					['page_size']: 5,
					['type']: 'not_connected',
					['language_code']: locale
				},
